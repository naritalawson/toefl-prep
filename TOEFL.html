    let currentSection = "";

    let score = 0;

    let total = 0;

    let timer;


    // -----------------------------

    // App Functions

    // -----------------------------

    function startSection(section) {

      currentSection = section;

      score = 0;

      total = questions[section].length;

      document.getElementById("result").innerText = "";

      loadQuestions(section);

      setTimer(section);

    }


    function loadQuestions(section) {

      const container = document.getElementById("quizContainer");

      container.style.display = "block";

      container.innerHTML = "";


      questions[section].forEach((q, index) => {

        let div = document.createElement("div");

        div.className = "question";


        if (q.audio) {

          div.innerHTML += `<audio controls><source src="${q.audio}" type="audio/mpeg"></audio>`;

        }

        if (q.passage) {

          div.innerHTML += `<div class="passage">${q.passage}</div>`;

        }


        div.innerHTML += `<p>${index + 1}. ${q.text}</p>`;

        q.options.forEach((opt, i) => {

          div.innerHTML += `<label><input type="radio" name="q${index}" value="${i}"> ${opt}</label><br>`;

        });


        container.appendChild(div);

      });


      let submitBtn = document.createElement("button");

      submitBtn.innerText = "Submit Section";

      submitBtn.onclick = checkAnswers;

      container.appendChild(submitBtn);

    }


    function checkAnswers() {

      let qList = questions[currentSection];

      score = 0;


      qList.forEach((q, index) => {

        let selected = document.querySelector(`input[name="q${index}"]:checked`);

        if (selected && parseInt(selected.value) === q.answer) {

          score++;

        }

      });


      clearInterval(timer);

      document.getElementById("timer").innerText = "";

      document.getElementById("result").innerText = 

        `Your score: ${score} / ${total}`;

    }


    function setTimer(section) {

      let time;

      if (section === "listening") time = 10 * 60;      // 10 minutes

      else if (section === "structure") time = 15 * 60; // 15 minutes

      else if (section === "reading") time = 20 * 60;   // 20 minutes


      timer = setInterval(() => {

        let minutes = Math.floor(time / 60);

        let seconds = time % 60;

        document.getElementById("timer").innerText =

          `Time remaining: ${minutes}:${seconds < 10 ? "0" : ""}${seconds}`;

        time--;


        if (time < 0) {

          clearInterval(timer);

          checkAnswers();

        }

      }, 1000);

    }
